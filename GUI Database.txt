CREATE TABLE ACCOUNT (
    ACCOUNTID VARCHAR(10) not null primary key,
    NAME VARCHAR(50),
    PHONENUMBER VARCHAR(20),
    EMAIL VARCHAR(50),
    USERNAME VARCHAR(50),
    PASSWORD VARCHAR(50),
    ROLE VARCHAR(10)
);

CREATE TABLE ORDERS (
    ORDERID VARCHAR(10) not null primary key,
    DATE DATE,
    ACCOUNTID VARCHAR(10),
    foreign key (ACCOUNTID) references ACCOUNT(ACCOUNTID)
);

CREATE TABLE PAYMENT (
    PAYID VARCHAR(10) not null primary key,
    PAYAMOUNT Decimal(19,2),
    STATSUS VARCHAR(10),
    DATE DATE,
    ORDERID VARCHAR(10),
    foreign key (ORDERID) references ORDERS(ORDERID)
);


CREATE TABLE CATEGORY (
    CATID VARCHAR(10) not null primary key,
    Name VARCHAR(50)
);

CREATE TABLE INVENTORY (
    INVENID VARCHAR(10) not null primary key,
    Name VARCHAR(50),
    PRICE Decimal(5,2),
    DESCRIPTION VARCHAR(100),
    PICTURE BLOB,
    DISCOUNT Decimal(6,2),
    CATID VARCHAR(10),
    foreign key (CATID) references CATEGORY(CATID)
);

CREATE TABLE ORDERDETAILS (
    ORDERID VARCHAR(10) not null,
    INVENID VARCHAR(10) not null,
    QUANTITY Decimal(10),
    PRIMARY KEY (ORDERID,INVENID),
    foreign key (ORDERID) references ORDERS(ORDERID),
    foreign key (INVENID) references INVENTORY(INVENID)
);
